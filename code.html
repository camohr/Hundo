                            function getValues() {
                            
                                let sv = parseInt (document.getElementById ("startValue").value);
                                let ev = parseInt (document.getElementById("endValue").value);
                            
                                if (!Number.isInteger(sv) || !Number.isInteger(ev)) {
                                    alert ("Please enter a number only.")
                                    return;
                                }
                            
                                let numbers = generateNumbers (sv, ev);
                                displayNumbers (numbers);
                            }
                            
                            // logic
                            function generateNumbers (start, end) {
                                if (end < start)  // ensure start less than end
                                {
                                    let m = end;
                                    end = start;
                                    start = m;
                                }
                                let numbers = [];
                                for (let i = start; i <= end; i++) {
                                    numbers.push(i);
                                }
                                return numbers;
                            }

Three functions


getValues()

Called by onclick event of btnSubmit in app.html. Reads the values that were entered in the start and end controls.

generateNumbers()

Creates a list of sequential numbers based on the start and end values.

displayNumbers()

Uses the list of numbers to create an array of table data elements. Injects the created data elements into the table's body.
